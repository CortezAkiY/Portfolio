void loop() {
  Usb.Task();



    rr = (PS3.getAnalogButton(R2) - PS3.getAnalogButton(L2)) / 2;         //Control with analog button

    if (PS3.getAnalogHat(LeftHatX) < 5 && PS3.getAnalogHat(LeftHatY) < 5) {
      xaxis = 0;
      yaxis = 0;

    } else {
      xaxis = PS3.getAnalogHat(LeftHatX) - 127.5;  //Control with joystic
      yaxis = (PS3.getAnalogHat(LeftHatY) - 127.5) * (-1);
    }


    

double V1 = xaxis + rr;                                            //omni1                                       
double V2 = xaxis*( -cos(pi / 3)) + yaxis*(sin(pi / 3)) + rr;      //omni2
double V3 = xaxis*(-cos(pi / 3)) + yaxis *(- sin(pi / 3)) + rr;    //omni3

    if (V1 > 0) {
      digitalWrite(D1, HIGH);
    } else if (V1 < 0) {

      V1 = V1 * (-1);
    }
    analogWrite(PWM1, V1);

    if (V2 > 0) {

      digitalWrite(D2, LOW);
      V2 = V2 * (-1);
    }
    analogWrite(PWM2, V2);

    if (V3 > 0) {

      digitalWrite(D3, LOW);
      V3 = V3 * (-1);
    }
    analogWrite(PWM3, V3);
